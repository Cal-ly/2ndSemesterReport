51 private Basket GetOrCreateBasketInternal(IHttpContextAccessor httpContextAccessor, ApplicationDbContext dbContext)
52 {
53 string? basketId = httpContextAccessor.HttpContext?.Request.Cookies["BasketId"];
54 Basket? basket;

56 if (string.IsNullOrEmpty(basketId))

58 basket = new Basket();
